196 __attribute__((always_inline)) __STATIC_INLINE uint32_t __RBIT(uint32_t value)
197 {
198  uint32_t result;
199  int32_t s = 4 /*sizeof(v)*/ * 8 - 1; // extra shift needed at end
200 
201  result = value; // r will be reversed bits of v; first get LSB of v
202  for (value >>= 1; value; value >>= 1)
203  {
204  result <<= 1;
205  result |= value & 1;
206  s--;
207  }
208  result <<= s; // shift when v's highest bits are zero
209  return(result);
210 }
